    <script type="text/javascript">
        function initialize() {

            var options = {types: ['(cities)'],componentRestrictions: {country: "ar"}};
            var input = document.getElementById('ciudad');
            var autocomplete = new google.maps.places.Autocomplete(input, options);

            google.maps.event.addListener(autocomplete, 'place_changed', function() {
                input.className = '';
                var place = autocomplete.getPlace();

                if (!place.geometry) {
                  // Inform the user that the place was not found and return.
                  input.className = 'notfound';
                  return;
                }

                $('#reference').val(place.reference);
                $("#ciudadError").html("");
            });
        }

        google.maps.event.addDomListener(window, 'load', initialize);

        (function () {

            $('#ciudad').on("propertychange input cut paste", function () {

                $("#ciudadError").html("<p>Comenz&aacute; a escribir y seleccion&aacute; del listado</p>");
                $("#reference").val("");
            });
        }());

        (function () {


            var validatorOptions = {
                custom: {

                    username: function($el) {

                        var pattern = '';

                        //caracteres permitidos
                        pattern = new RegExp(/^([a-z0-9\u00f1_-])+$/i);

                        if(!pattern.test($el.val())) {
                            return decodeHtml('<?= $this->lang->line('username')?>');
                        }

                        //debe comenzar por letra
                        pattern = new RegExp(/^[a-z]/i);

                        if(!pattern.test($el.val())) {
                            return decodeHtml('<?= $this->lang->line('first_letter')?>');
                        }

                        //limitar como se usan los guiones
                        pattern = new RegExp(/^((?!.*_-)(?!.*-_)(?!.*__)(?!.*--)).*$/);

                        if(!pattern.test($el.val())) {
                            return decodeHtml('<?= $this->lang->line('limit_scores')?>');
                        }

                        //caracteres como minimo
                        pattern = new RegExp(/.{<?= $this->config->item('username_min_length', 'tank_auth')?>,}/);

                        if(!pattern.test($el.val())) {

                            return decodeHtml('<?= sprintf($this->lang->line('min_length'), '', $this->config->item('username_min_length', 'tank_auth')); ?>');
                        }

                        //debe terminar con alfanumerico
                        pattern = new RegExp(/[a-z0-9\u00f1]$/i);

                        if(!pattern.test($el.val())) {
                            return 'e';
                        }

                    }
                }
            };

            $("#registerForm").validator(validatorOptions);

        }());

        (function () {
            $( "#registerForm" ).submit(function( event ) {
                $("#submit").prop("disabled",true);
                $("#submit").val("Espere ...");

                if($('#reference').val()==""){
                    $("#ciudad").val("");
                }
            })
        }());
